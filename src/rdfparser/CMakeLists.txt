SET (subproject_SOURCES
  alternatelinkreader.cpp
  raptorparser.cpp
  rssparserdialog.cpp
  rsstreeview.cpp
  rssviewer.cpp
  rsshighlighter.cpp
)

SET (subproject_HEADERS
  alternatelinkreader.h
  raptorparser.h
  rssparserdialog.h
  rsstreeview.h
  rssviewer.h
  rsshighlighter.h
)

IF (ACCEPT_MICROSOFT_PUBLIC_LICENSE)
SET (subproject_SOURCES
  ${subproject_SOURCES}
  xsdparser.cpp
)

  ${subproject_HEADERS}
  xsdparser.h
)
SET (XML_PATTERNS_LIBRARY
  ${QT_QTXMLPATTERNS_LIBRARY}
)
ENDIF (ACCEPT_MICROSOFT_PUBLIC_LICENSE)

INCLUDE_DIRECTORIES (${XHTMLDBG_BINARY_PROJECT_INCULDE})

REMOVE_DEFINITIONS (-DQT_DBUS_LIB)

  ADD_DEFINITIONS (-DMS_PL_ACCEPTED)
ENDIF (ACCEPT_MICROSOFT_PUBLIC_LICENSE)

QT4_WRAP_CPP (subproject_MOC ${subproject_HEADERS})

QT4_ADD_RESOURCES (subproject_RESRC rdfparser.qrc)

ADD_LIBRARY (RDFParser STATIC
  ${subproject_MOC}
  ${subproject_HEADERS}
  ${subproject_RESRC}
  ${subproject_SOURCES}
)

TARGET_LINK_LIBRARIES (RDFParser
  ${XHTMLDBG_LIBRARIES}
  ${XML_PATTERNS_LIBRARY}
  Networker
)
