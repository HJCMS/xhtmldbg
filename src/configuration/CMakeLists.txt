SET (subproject_SOURCES
  configuration.cpp
  configurationmenu.cpp
  stackedwidget.cpp
  pagewidget.cpp
  configide.cpp
  configtidy.cpp
  configdomtree.cpp
  configbrowser.cpp
  configcookies.cpp
  configproxy.cpp
  configssl.cpp
  configtrustedhosts.cpp
  configaccesscontrol.cpp
  configuseragents.cpp
  configgeo.cpp
  configutils.cpp
)

SET (subproject_HEADERS
  configuration.h
  configurationmenu.h
  stackedwidget.h
  pagewidget.h
  configide.h
  configtidy.h
  configdomtree.h
  configbrowser.h
  configcookies.h
  configproxy.h
  configssl.h
  configtrustedhosts.h
  configaccesscontrol.h
  configuseragents.h
  configgeo.h
  configutils.h
)

  ${CMAKE_CURRENT_SOURCE_DIR}

  ${CMAKE_CURRENT_SOURCE_DIR}/useragenteditor
  ${CMAKE_CURRENT_BINARY_DIR}/certissuers
  ${CMAKE_CURRENT_SOURCE_DIR}/certissuers

)

  ${subproject_MOC}
  ${subproject_HEADERS}
  ${subproject_SOURCES}
)

SET_TARGET_PROPERTIES (Configuration PROPERTIES
  LINKER_LANGUAGE CXX
  COMPILE_FLAGS "${LIB_FPIC_CXXFLAGS}"
)

TARGET_LINK_LIBRARIES (Configuration
  ${QT_QTCORE_LIBRARY}
  ${QT_QTGUI_LIBRARY}
  EditCookiesTable
  UserAgentsEditor
  CertIssuers
)
