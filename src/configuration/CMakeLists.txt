  configuration.cpp
  configurationmenu.cpp
  stackedwidget.cpp
  featurebox.cpp
  pagewidget.cpp
  configide.cpp

  configuration.h
  configurationmenu.h
  stackedwidget.h
  featurebox.h
  pagewidget.h
  configide.h
  configtidy.h
  configdomtree.h

  configproxy.h

  ${CMAKE_CURRENT_SOURCE_DIR}/directorychooser

)

  ${configuration_MOC}
  ${configuration_HEADERS}
  ${configuration_SOURCES}
)

  LINKER_LANGUAGE CXX
  COMPILE_FLAGS "${LIB_FPIC_CXXFLAGS}"
)

  ${QT_QTCORE_LIBRARY}
