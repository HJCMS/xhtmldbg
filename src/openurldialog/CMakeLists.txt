# CMakeLists.txt

SET (openurldialog_SOURCES
  openurldialog.cpp
)

SET (openurldialog_HEADERS
  openurldialog.h
)

REMOVE_DEFINITIONS (-DQT_DBUS_LIB -DQT_WEBKIT_LIB -DQT_XML_LIB -DQT_XMLPATTERNS_LIB)

QT4_WRAP_CPP (openurldialog_MOC ${openurldialog_HEADERS})

ADD_DEFINITIONS (${QTIDY_DEFINITIONS})

INCLUDE_DIRECTORIES (
  ${CMAKE_CURRENT_BINARY_DIR}
  ${XHTMLDBG_BINARY_PROJECT_INCULDE}
)

ADD_LIBRARY (OpenUrlDialog STATIC
  ${openurldialog_MOC}
  ${openurldialog_HEADERS}
  ${openurldialog_SOURCES}
)

SET_TARGET_PROPERTIES (OpenUrlDialog PROPERTIES
  LINKER_LANGUAGE CXX
  COMPILE_FLAGS "${LIB_FPIC_CXXFLAGS}"
)

TARGET_LINK_LIBRARIES (OpenUrlDialog
  ${QT_QTMAIN_LIBRARY}
  ${QT_QTCORE_LIBRARY}
  ${QT_QTGUI_LIBRARY}
)
