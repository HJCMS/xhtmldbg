# CMakeLists.txt

SET (settings_SOURCES
  settings.cpp
  websettings.cpp
)

SET (settings_HEADERS
  settings.h
  websettings.h
)

SET (settings_NOMOC_HEADERS
)

INCLUDE_DIRECTORIES (
  ${XHTMLDBG_BINARY_PROJECT_INCULDE}
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}
)

QT4_WRAP_CPP (settings_MOC ${settings_HEADERS})

ADD_LIBRARY (Settings STATIC
  ${settings_MOC}
  ${settings_HEADERS}
  ${settings_NOMOC_HEADERS}
  ${settings_SOURCES}
)

REMOVE_DEFINITIONS (-DQT_DBUS_LIB -DQT_XML_LIB -DQT_XMLPATTERNS_LIB)

SET_TARGET_PROPERTIES (Settings PROPERTIES
  LINKER_LANGUAGE CXX
  COMPILE_FLAGS "${LIB_FPIC_CXXFLAGS}"
)

TARGET_LINK_LIBRARIES (Settings
  ${QT_QTCORE_LIBRARY}
  ${QT_QTGUI_LIBRARY}
  ${QT_QTWEBKIT_LIBRARY}
)
